然后在js中引用anime动画库

const captionsEl = document.querySelectorAll(".slide-caption"); //获取实例对象

glide.on(["mount.after","run.after"],() => {
    // 返回当前轮播的索引（下标）
    const caption = captionsEl[glide.index]
    anime({
        targets: caption.children,
        opacity: [0,1],
        // 透明度是从0 到 1 
        duration: 400,
        // 动画执行时间，400ms
        easing:"linear",
        // 动画执行函数,  linear线性 也就是平滑执行
        delay:anime.stagger(400,{start: 300}),
        // anime.stagger()设定交错开始值, 延迟第一个元素从300ms开始，然后之后每隔元素元素增加400ms
        translateY: [anime.stagger([40,10]),0]
         // 从y轴出现的距离，40到10，第一个最长，最后一个最短，最终到0  自己的位置
    });
});
//做初始化为每次进入页面都有动画
glide.on("run.before",() => {
    document.querySelectorAll(".slide-caption > * ").forEach(el => {
        el.style.opacity = 0
    })

自动筛选功能

要实现点击按钮会自动筛选，所以要给页面定义一下类别

<button class="filter-btn active" data-filter="*">全部</button> <button class="filter-btn" data-filter=".web">WFB</button> <button class="filter-btn" data-filter=".mobile">移动</button> <button class="filter-btn" data-filter=".science">科研</button>

web science mobile上面代码已经添加

然后在js中添加代码实现动画点击图片效果

const filterBtns = document.querySelector('.filter-btns');
// 实现点击自动换
filterBtns.addEventListener("click", e => {
    let {target} = e;
    const filterOption = target.getAttribute("data-filter");
    if (filterOption) {
        document.querySelectorAll(".filter-btn.active").forEach(btn => {
            btn.classList.remove("active");
            target.classList.add("active");

        isotope.arrange({ filter: filterOption});
        })
    }
});

const headerEl = document.querySelector("header");
const scrollToTop = document.querySelector(".scrollToTop") 
window.addEventListener("scroll",() => {
    // 本身的高度
    let height = headerEl.getBoundingClientRect().height;
    if (window.pageYOffset - height > 800) {
        if (!headerEl.classList.contains("sticky")) {
            headerEl.classList.add("sticky");
        }

    }else{
        headerEl.classList.remove("stickcy");
       }
       if (window.pageYOffset > 2000) {
           scrollToTop.style.display = "block";
       }else{
        scrollToTop.style.display = "none";
       }
})

第一眼就被这个网站惊艳到。这个企业网站包含了很多动态特效其中有一些用到了css3新特性实现特效效果，而且了解grid这种布局模式。可以跟flex进行对比，grid是相对于三维布局.flex只是一维布局。今天学到了什么知识点：vscode的快捷写.加上对应的类名.类名可以同时添加 当然.表示的是div 也可以自定义。学会了用section作为块级把每个功能写在一个section中，样式也是一个大盒子包含着小盒子。然后就是一些动画特效，封装的组件库等等，使用js进行调用拿到节点进行操作，还不是很熟悉等进行加强。

使用了h5的语义化标签；
